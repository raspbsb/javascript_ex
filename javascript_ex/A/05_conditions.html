      }
      조건 - 비교연산자
      > < >= <= == === != !==
      0 == false
      1 == true
      null, undefined == false

타입 비교

if else

      if(조건){
        참일 때 실행
      } else{
        거짓일 때 실행
      }

중첩 조건문

      if(조건1){
        if(조건2){
          if(조건3){
            
          }
        }
      }

조건문의 축약

      축약1 - 중괄호 축약
      if(조건)
      else
      ;

      축약2 - 삼항연산자 (반드시 else가 있는 구문에만 사용)
      조건? 참일때 실행 : 거짓일 때 실행

      축약3 - 논리연산자
      조건 && 참일 때 실행할 코드

논리연산자

      조건1 && 조건2 : and. 두 조건이 모두 참이어야 결과가 참
      조건1 || 조건2 : or. 두 조건 중 하나라도 참이면 결과가 참

switch

      switch(조건){
        case 값1:
          값 1일때 실행
          break; //종료
        case 값2:
          값 2일때 실행
          break; //종료
        case 값3:
          값 3일때 실행
          break; //종료
        default: 
          기본일 때 할 일
          break
      }
